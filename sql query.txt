


CREATE TABLE RestaurantProducts ( ProductID INT PRIMARY KEY IDENTITY(1000,1), ProductName NVARCHAR(100) NOT NULL, ProductPrice DECIMAL(10, 2) NOT NULL, ProductPicture VARBINARY(MAX) NULL, RestaurantID INT NOT NULL, FOREIGN KEY (RestaurantID) REFERENCES Restaurant(rID) );

CREATE TABLE Admin ( A_ID INT PRIMARY KEY IDENTITY(100,1), A_Email NVARCHAR(100) NOT NULL UNIQUE, A_Pass NVARCHAR(100) NOT NULL );

CREATE TABLE Restaurant ( rID INT PRIMARY KEY IDENTITY(1,1), rName NVARCHAR(100) NOT NULL, rEmail NVARCHAR(100) NOT NULL UNIQUE, rPass NVARCHAR(100) NOT NULL, ImageData VARBINARY(MAX) NULL, rDetails NVARCHAR(MAX) NULL; );

CREATE TABLE OrderDetails ( OrderDetailID INT PRIMARY KEY IDENTITY(1,1), OrderID INT FOREIGN KEY REFERENCES [Order](OrderID), ProductID INT FOREIGN KEY REFERENCES RestaurantProducts(ProductID), Quantity INT, Price DECIMAL(10, 2) );

CREATE TABLE [Order] ( OrderID INT PRIMARY KEY IDENTITY(100,1), CustomerID INT FOREIGN KEY REFERENCES Customer(C_ID), RestaurantID INT FOREIGN KEY REFERENCES Restaurant(rID), OrderDate DATETIME, TotalPrice DECIMAL(10, 2), OrderStatus NVARCHAR(50) NOT NULL DEFAULT 'Pending' );

CREATE TABLE Customer (C_ID INT IDENTITY(1,1) PRIMARY KEY, C_Name NVARCHAR(50) NOT NULL, C_Password NVARCHAR(50) NOT NULL, C_Email NVARCHAR(50) NOT NULL, C_Gender NVARCHAR(10) NOT NULL);


CREATE TABLE RestaurantTheme (
    tPic VARBINARY(MAX) NULL,
    tID INT NOT NULL,
    PRIMARY KEY (tID)
);

CREATE TABLE RestaurantTheme (
    tID INT IDENTITY(1,1) PRIMARY KEY,
    tPic VARBINARY(MAX) NULL
);
